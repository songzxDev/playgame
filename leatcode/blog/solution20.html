/**
 * 题目：20.有效的括号
 * 标签：字符串 栈
 * @param {string} s
 * @return {boolean}
 */
const isValid = (str) => {
    let stack = [];
    for (let s of str) {
        if (s === '(') {
            stack.push(')');
        } else if (s === '[') {
            stack.push(']');
        } else if (s === '{') {
            stack.push('}');
        } else if (stack.length === 0 || stack.pop() !== s) {
            return false;
        }
    }
    return stack.length === 0;
};
